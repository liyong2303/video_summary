```vue
<script setup lang="ts">
import { ref, onMounted, watch } from 'vue'
import mermaid from 'mermaid'
import { ElMessage } from 'element-plus'

interface Props {
  mermaidCode: string
}

const props = defineProps<Props>()

const containerRef = ref<HTMLElement>()
const isLoading = ref(true)

onMounted(() => {
  mermaid.initialize({ startOnLoad: false })
  renderMindmap()
})

watch(() => props.mermaidCode, () => {
  renderMindmap()
})

async function renderMindmap() {
  if (!containerRef.value || !props.mermaidCode) return

  isLoading.value = true
  try {
    const { svg } = await mermaid.render('mermaid-' + Date.now(), props.mermaidCode)
    containerRef.value.innerHTML = svg
  } catch (error) {
    console.error('Failed to render mindmap:', error)
    ElMessage.error('渲染思维导图失败')
  } finally {
    isLoading.value = false
  }
}

function copyCode() {
  navigator.clipboard.writeText(props.mermaidCode)
  ElMessage.success('已复制 Mermaid 代码')
}

function exportImage() {
  // 简单实现：使用浏览器打印功能
  const printWindow = window.open('', '_blank')
  if (printWindow) {
    printWindow.document.write(`
      <html>
        <head><title>思维导图</title></head>
        <body style="display:flex;justify-content:center;align-items:center;min-height:100vh;margin:0;">
          ${containerRef.value?.innerHTML || ''}
        </body>
      </html>
    `)
    printWindow.document.close()
    printWindow.print()
  }
}
</script>

<template>
  <div class="mindmap-renderer">
    <div class="toolbar">
      <el-button size="small" @click="copyCode">复制代码</el-button>
```

```vue
<script setup lang="ts">
import { ref, computed } from 'vue'

interface Props {
  content: string
}

const props = defineProps<Props>()

interface Scene {
  id: number
  title: string
  duration: string
  shot: string
  content: string
}

const scenes = computed<Scene[]>(() => {
  if (!props.content) return []

  // 解析脚本内容
  const lines = props.content.split('\n')
  const result: Scene[] = []
  let currentScene: Partial<Scene> | null = null

  lines.forEach((line, index) => {
    const sceneMatch = line.match(/【场景(\d+)】(.+?)\s*-\s*时长：(.+)/)
    if (sceneMatch) {
      if (currentScene) {
        result.push(currentScene as Scene)
      }
      currentScene = {
        id: parseInt(sceneMatch[1]),
        title: sceneMatch[2],
        duration: sceneMatch[3],
        content: ''
      }
    } else if (currentScene) {
      const shotMatch = line.match(/镜头：(.+)/)
      const subtitleMatch = line.match(/字幕：(.+)/)
      const narratorMatch = line.match(/旁白：(.+)/)

      if (shotMatch) {
        currentScene.shot = shotMatch[1]
      } else if (subtitleMatch || narratorMatch) {
        currentScene.content += (subtitleMatch || narratorMatch)![1] + '\n'
      } else if (line.trim()) {
        currentScene.content += line + '\n'
      }
    }
  })

  if (currentScene) {
    result.push(currentScene as Scene)
  }

  return result
})
</script>

<template>
  <div class="script-viewer">
    <div v-if="scenes.length === 0" class="empty">
      暂无脚本内容
    </div>
    <div v-else class="scenes">
      <div v-for="scene in scenes" :key="scene.id" class="scene-card">
        <div class="scene-header">
          <span class="scene-title">【场景{{ scene.id }}】{{ scene.title }}</span>
          <el-tag size="small">{{ scene.duration }}</el-tag>
        </div>
        <div v-if="scene.shot" class="scene-shot">
          <span class="label">镜头：</span>
          <span>{{ scene.shot }}</span>
        </div>
        <div v-if="scene.content" class="scene-content">
          {{ scene.content }}
        </div>
      </div>
    </div>
  </div>
</template>
```

```sql
    id BIGINT AUTO_INCREMENT PRIMARY KEY,

    enabled BOOLEAN DEFAULT TRUE COMMENT '是否启用',
    style VARCHAR(20) NOT NULL DEFAULT 'professional' COMMENT '文案风格',
    min_length INT DEFAULT 100 COMMENT '最小字数',
    max_length INT DEFAULT 500 COMMENT '最大字数',
    tags TEXT COMMENT '建议标签（JSON数组）',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY uk_user_platform (user_id, platform),
    INDEX idx_user_id (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 多平台文案输出表
CREATE TABLE IF NOT EXISTS social_output (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,

    platform VARCHAR(50) NOT NULL COMMENT '平台名称',

    INDEX idx_user_id (user_id),
```
