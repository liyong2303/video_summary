## 架构分析

### 现状
- **Prompt 定义**: `video-summary-ai/app/prompts.py` 定义了 4 个 prompt 函数，风格固定
- **AI 服务接口**: `PipelineRequest` 和 `SingleStepRequest` 只接受 `task_id` 和 `subtitle_text`
- **数据库**: `user` 表只存储用户名、密码、角色，无配置信息
- **前端**: 提交页面只有 URL 输入框，无风格和字数选择

### 需要的变更
1. **数据库**: 新增 `user_preference` 表（存储字数偏好、输出格式）和 `custom_prompt` 表（用户自定义 Prompt）
2. **Python AI 服务**: 
   - 修改 prompts.py 支持 3 种内置风格
   - 修改 PipelineRequest/SingleStepRequest 接受 `style` 和 `length` 参数
   - 实现自定义 Prompt 加载逻辑
3. **Java 后端**:
   - 创建 `UserPreference` 和 `CustomPrompt` 实体和 Mapper
   - 创建 `UserPreferenceService` 管理用户配置
   - 修改 `PipelineClient` 传递风格和字数参数
4. **前端**:
   - 提交页面添加风格选择和字数选择
   - 创建设置页面（偏好设置、自定义 Prompt 管理）

#### 2.1 更新 `prompts.py` 添加风格变体

**新增 3 种风格的 system prompt**:

```python
STYLE_SYSTEM_PROMPTS = {
    "academic": "你是一个专业的学术研究者，擅长用严谨、客观、有逻辑的语言总结内容。",
    "casual": "你是一个轻松友好的内容创作者，擅长用口语化、生动有趣的语言表达观点。",
    "concise": "你是一个干货提炼专家，擅长用最简洁的语言传递核心信息。"
}

LENGTH_INSTRUCTIONS = {
    "short": "控制输出长度为标准的50%左右，只保留最核心的信息。",
    "standard": "正常输出，平衡详细度和简洁性。",
    "long": "扩展输出长度为标准的150%，添加更多细节、案例和背景信息。"
}

def summary_prompt(subtitle_text: str, style: str = "concise", length: str = "standard") -> tuple[str, str]:
    """Step 1: Generate summary from subtitle text."""
    system = STYLE_SYSTEM_PROMPTS.get(style, STYLE_SYSTEM_PROMPTS["concise"])
    system += " " + LENGTH_INSTRUCTIONS.get(length, LENGTH_INSTRUCTIONS["standard"])

    prompt = f"""请根据以下视频字幕内容，生成一份结构化的视频总结。

要求：
1. 列出3-5个核心观点，每个观点附带时间戳（如果字幕中有时间信息）
2. 最后用一段话总结视频的核心思想
3. 语言简洁有力，避免冗余
4. 保留关键数据和专有名词

字幕内容：
{subtitle_text}"""
    return system, prompt
```

```java
package com.videosummary.service;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.videosummary.entity.UserPreference;
import com.videosummary.mapper.UserPreferenceMapper;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.util.Arrays;
import java.util.List;

@Service
@RequiredArgsConstructor
public class UserPreferenceService {

    private final UserPreferenceMapper userPreferenceMapper;

    /**
     * 获取用户偏好，如果不存在则返回默认值
     */
    public UserPreference getOrDefault(Long userId) {
        UserPreference pref = userPreferenceMapper.selectOne(
                new LambdaQueryWrapper<UserPreference>().eq(UserPreference::getUserId, userId)
        );
        if (pref == null) {
            pref = UserPreference.builder()
                    .userId(userId)
                    .style("concise")
                    .length("standard")
                    .outputTypes(Arrays.asList("summary", "article", "card", "xiaohongshu"))
                    .build();
        }
        return pref;
    }

    /**
     * 保存或更新用户偏好
     */
    public void saveOrUpdate(UserPreference preference) {
        UserPreference existing = userPreferenceMapper.selectOne(
                new LambdaQueryWrapper<UserPreference>()
                        .eq(UserPreference::getUserId, preference.getUserId())
        );
        if (existing != null) {
            preference.setId(existing.getId());
            userPreferenceMapper.updateById(preference);
        } else {
            userPreferenceMapper.insert(preference);
        }
    }
}
```

#### 3.4 创建 `CustomPromptService`

**路径**: `video-summary-service/src/main/java/com/videosummary/service/CustomPromptService.java`

```java
package com.videosummary.service;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.videosummary.entity.CustomPrompt;
import com.videosummary.mapper.CustomPromptMapper;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.util.List;

@Service
@RequiredArgsConstructor
public class CustomPromptService {

    private final CustomPromptMapper customPromptMapper;

    /**
     * 获取用户的自定义 Prompt 列表
     */
    public List<CustomPrompt> getByUserId(Long userId) {
        return customPromptMapper.selectList(
                new LambdaQueryWrapper<CustomPrompt>()
                        .eq(CustomPrompt::getUserId, userId)
                        .orderByDesc(CustomPrompt::getCreatedAt)
        );
    }

    /**
     * 获取用户指定类型的默认自定义 Prompt
     */
    public CustomPrompt getDefaultByUserAndType(Long userId, String outputType) {
        return customPromptMapper.selectOne(
                new LambdaQueryWrapper<CustomPrompt>()
                        .eq(CustomPrompt::getUserId, userId)
                        .eq(CustomPrompt::getOutputType, outputType)
                        .eq(CustomPrompt::getIsDefault, true)
        );
    }

    /**
     * 保存自定义 Prompt
     */
    public void save(CustomPrompt prompt) {
        customPromptMapper.insert(prompt);
    }

    /**
     * 更新自定义 Prompt
     */
    public void update(CustomPrompt prompt) {
        customPromptMapper.updateById(prompt);
    }

    /**
     * 删除自定义 Prompt
     */
    public void delete(Long id, Long userId) {
        CustomPrompt prompt = customPromptMapper.selectById(id);
        if (prompt == null || !prompt.getUserId().equals(userId)) {
            throw new IllegalArgumentException("Prompt不存在或无权删除");
        }
        customPromptMapper.deleteById(id);
    }

    /**
     * 设置默认 Prompt
     */
    public void setDefault(Long id, Long userId, String outputType) {
        // 清除该用户该类型的所有默认标记
        customPromptMapper.update(null,
                new LambdaQueryWrapper<CustomPrompt>()
                        .eq(CustomPrompt::getUserId, userId)
                        .eq(CustomPrompt::getOutputType, outputType)
                        .eq(CustomPrompt::getIsDefault, true),
                new com.baomidou.mybatisplus.core.update.LambdaUpdateWrapper<CustomPrompt>()
                        .set(CustomPrompt::getIsDefault, false)
        );
        // 设置新的默认
        CustomPrompt prompt = customPromptMapper.selectById(id);
        if (prompt != null && prompt.getUserId().equals(userId)) {
            prompt.setIsDefault(true);
            customPromptMapper.updateById(prompt);
        }
    }
}
```

### Task 8: 前端 — 自定义 Prompt 管理（高级功能）

**路径**: `video-summary-frontend/src/views/CustomPromptView.vue`

```vue
<script setup lang="ts">
import { ref, onMounted } from 'vue'
import axios from 'axios'
import { ElMessage } from 'element-plus'

interface CustomPrompt {
  id?: number
  name: string
  outputType: string
  systemPrompt: string
  userPrompt: string
  isDefault: boolean
}

const prompts = ref<CustomPrompt[]>([])
const loading = ref(false)
const dialogVisible = ref(false)
const editingPrompt = ref<CustomPrompt>({
  name: '',
  outputType: 'summary',
  systemPrompt: '',
  userPrompt: '',
  isDefault: false,
})

const outputTypeOptions = [
  { label: '总结', value: 'summary' },
  { label: '文章', value: 'article' },
  { label: '学习卡片', value: 'card' },
  { label: '小红书文案', value: 'xiaohongshu' },
]

async function fetchPrompts() {
  loading.value = true
  try {
    const res = await axios.get('/api/user/custom-prompts')
    if (res.data.code === 0) {
      prompts.value = res.data.data
    }
  } catch (e) {
    console.error('Failed to fetch prompts', e)
  } finally {
    loading.value = false
  }
}

async function savePrompt() {
  try {
    if (editingPrompt.value.id) {
      await axios.put(`/api/user/custom-prompts/${editingPrompt.value.id}`, editingPrompt.value)
    } else {
      await axios.post('/api/user/custom-prompts', editingPrompt.value)
    }
    ElMessage.success('保存成功')
    dialogVisible.value = false
    fetchPrompts()
  } catch (e: any) {
    ElMessage.error(e.response?.data?.message || '保存失败')
  }
}

async function deletePrompt(id: number) {
  try {
    await axios.delete(`/api/user/custom-prompts/${id}`)
    ElMessage.success('删除成功')
    fetchPrompts()
  } catch (e: any) {
    ElMessage.error(e.response?.data?.message || '删除失败')
  }
}

async function setDefault(id: number, outputType: string) {
  try {
    await axios.post(`/api/user/custom-prompts/${id}/set-default`, { outputType })
    ElMessage.success('设置成功')
    fetchPrompts()
  } catch (e: any) {
    ElMessage.error(e.response?.data?.message || '设置失败')
  }
}

function openDialog(prompt?: CustomPrompt) {
  editingPrompt.value = prompt ? { ...prompt } : {
    name: '',
    outputType: 'summary',
    systemPrompt: '',
    userPrompt: '',
    isDefault: false,
  }
  dialogVisible.value = true
}

onMounted(() => {
  fetchPrompts()
})
</script>

<template>
  <div class="custom-prompt-view">
    <h2 class="page-title">自定义 Prompt</h2>

    <el-button type="primary" @click="openDialog()">
      <el-icon><Plus /></el-icon> 新建 Prompt
    </el-button>

    <el-table v-loading="loading" :data="prompts" style="margin-top: 24px">
      <el-table-column prop="name" label="名称" width="200" />
      <el-table-column prop="outputType" label="类型" width="120">
        <template #default="{ row }">
          {{ outputTypeOptions.find(o => o.value === row.outputType)?.label || row.outputType }}
        </template>
      </el-table-column>
      <el-table-column prop="systemPrompt" label="System Prompt" show-overflow-tooltip />
      <el-table-column prop="isDefault" label="默认" width="80">
        <template #default="{ row }">
          <el-tag v-if="row.isDefault" type="success" size="small">是</el-tag>
          <el-tag v-else type="info" size="small">否</el-tag>
        </template>
      </el-table-column>
      <el-table-column label="操作" width="180">
        <template #default="{ row }">
          <el-button type="primary" link size="small" @click="openDialog(row)">编辑</el-button>
          <el-button
            v-if="!row.isDefault"
            type="success"
            link
            size="small"
            @click="setDefault(row.id!, row.outputType)"
          >
            设为默认
          </el-button>
          <el-button type="danger" link size="small" @click="deletePrompt(row.id!)">删除</el-button>
        </template>
      </el-table-column>
    </el-table>

    <el-dialog v-model="dialogVisible" :title="editingPrompt.id ? '编辑 Prompt' : '新建 Prompt'" width="600px">
      <el-form :model="editingPrompt" label-width="100px">
        <el-form-item label="名称">
          <el-input v-model="editingPrompt.name" placeholder="请输入名称" />
        </el-form-item>
        <el-form-item label="类型">
          <el-select v-model="editingPrompt.outputType">
            <el-option
              v-for="opt in outputTypeOptions"
              :key="opt.value"
              :label="opt.label"
              :value="opt.value"
            />
          </el-select>
        </el-form-item>
        <el-form-item label="System Prompt">
          <el-input
            v-model="editingPrompt.systemPrompt"
            type="textarea"
            :rows="4"
            placeholder="请输入 System Prompt"
          />
        </el-form-item>
        <el-form-item label="User Prompt">
          <el-input
            v-model="editingPrompt.userPrompt"
            type="textarea"
            :rows="6"
            placeholder="请输入 User Prompt 模板，使用 {content} 作为内容占位符"
          />
        </el-form-item>
        <el-form-item label="设为默认">
          <el-switch v-model="editingPrompt.isDefault" />
        </el-form-item>
      </el-form>
      <template #footer>
        <el-button @click="dialogVisible = false">取消</el-button>
        <el-button type="primary" @click="savePrompt">保存</el-button>
      </template>
    </el-dialog>
  </div>
</template>
```
